  4 /**
  5  * Creates an instance of Node.
  6  *
  7  * @constructor
  8  * @this {Node}
  9  * @param {string} value Label of the element.
 10  * @param {number} id Unique id of the element. Use a negative value to indicate root element
 11  * @param {string} urlSubItems Url to ajax load.
 12  */
 13 JusTo.ui.Node = function(value, id, urlSubItems) {
 14     /** @private */ this.parent = null;
 15     /** @private */ this.children = [];
 16     /** @private */ this.childDom = null;
 17     this.value = value || "";
 18     this.id = id;
 19     this.selected = false;
 20     /** @private */ this.visible = true;
 21     /** @private */ this.childrenVisible = urlSubItems === undefined;
 22     /** @private */ this.urlSubItems = urlSubItems;
 23     /** @private */ this.options = {};
 24 
 25     /** @private */ this.innerDoms = {};
 26     /** @private */ this.outterDom = this._newListItem(this);
 27 
 28     this.events.init();
 29 };

 31 /**
 32  * An attribute to store the events.
 33  *
 34  * @return {Object} Events.
 35  */
 36 JusTo.ui.Node.prototype.events = (function() {
 37 // TODO move this method to bind only with root node. With this implementation
 38 // it is allowed only one tree instance per document
 39     return {
 40         initialized: false,
 41         init: function() {
 42             if (this.initialized)
 43                 return;
 44 
 45             //TODO this function bind needs to be placed outside event scope
 46             JusTo.ui.Node.canSelect = JusTo.ui.NodeFilter.any;
 47 
 48             this.initOpenState();
 49             this.initNodeTypeChanged();
 50             this.initContentClicked();
 51 
 52             this.initialized = true;
 53         },
 54         initOpenState: function() {
 55             this.openStateChanged.push(function(node) {
 56                 var iconDom = node.innerDoms.icon;
 57                 iconDom.className = "icon";
 58                 iconDom.className += node.childrenVisible ? " folder-opened" : " folder-closed";
 59             });
 60         },
 61         initNodeTypeChanged: function() {
 62             this.nodeTypeChanged.push(function(node) {
 63                 node._nodeOpenChanged(node);
 64             });
 65         },
 66         initContentClicked: function() {
 67             this.contentClicked.push(function(evt, node) {
 68                 node.setSelected(!node.selected);
 69             });
 70         },
 71         openStateChanged: [], //occurs when node toggle his children visibility
 72         nodeTypeChanged: [],  //occurs when node change from leaf to non-leaf
 73         selectionChanged: [], //occurs when the node was (de)selected
 74         contentClicked: [],   //occurs when the user click at content span
 75         asyncOpened: []       //occurs when the node was opened in async mode
 76     };

 89 /**
 90  * Returns the node found or null if not found.
 91  *
 92  * @this {Node}
 93  * @return {Node} The node found.
 94  */
 95 JusTo.ui.Node.prototype.find = function(nodeId) {
 96     var found = null;
 97     var result = this.iterate(function(node, depth) {
 98         if (node.id === nodeId) {
 99             found = node;
100             return false;
101         }
102         return true;
103     });
104     return found;
105 };

107 JusTo.ui.Node.prototype.findRoot = function() {
108     var parent = this.parent;
109     while(parent !== null) {
110         parent = parent.parent;
111     }
112     return parent;
113 };

216 JusTo.ui.Node.prototype._insertNode = function(node) {
217     var parentElement = this.outterDom;
218     if (!this.isRoot()) {
219         if (!this.hasChildren()) {
220             this.childDom = document.createElement("ul");
221             parentElement.appendChild(this.childDom);
222             this._setChildrenVisible(true);
223         }
224 
225         parentElement = this.childDom;
226         this.outterDom.firstChild.className = "expander expanded";
227         this.outterDom.firstChild.style.display = '';
228     }
229     parentElement.appendChild(node.outterDom);
230 };

288 /**
289  * Opens a node. 
290  * This method will automatically check if exist URL to load data via ajax, 
291  * once data was loaded, data will be cached at memory
292  * Examples: node.open(), node.open(50), node.open(10, 20, 30, 40),
293  * node.open([10, 20, 30, 40])
294  *
295  * @this {Node}
296  * @see events.openStateChanged
297  * @param {Object=} optional This param can be omitted or multiple
298  */
299 JusTo.ui.Node.prototype.open = function() {
300     switch (arguments.length) {
301         case 0: {
302             return this._open();
303         } case 1: {
304             if(typeof(arguments[0]) == 'number') {
305                 return this._openId(arguments[0]);
306             }
307             return this._openPath(Array.prototype.slice.call(arguments[0]));
308         }
309     }
310     return this._openPath(Array.prototype.slice.call(arguments));
311 };
312 
313 JusTo.ui.Node.prototype._open = function(nodeId) {
314     if (this.urlSubItems !== undefined && this.loaded !== true) {
315         this._asyncOpen();
316     } else {
317         this.innerDoms.expander.className = 'expander expanded';
318 
319         this._setChildrenVisible(true);
320         this._nodeOpenChanged(this);
321     }
322     return this;
323 };

325 JusTo.ui.Node.prototype._openId = function(nodeId) {
326     var node = this.find(nodeId);
327     if(node !== null) {
328         node.open();
329         return node;
330     }
331     return null;
332 };

334 JusTo.ui.Node.prototype._openPath = function(path) {
335     var depth = -1;
336 
337     var evt = function(node) {  
338         if(!node.isOpened()) return;
339         depth++;
340         var id = path[depth];
341 
342         var child = node.find(id);
343         if(depth === path.length - 1) {
344             if(child.isLeaf()) {
345                 child.setSelected(true);
346             } else {
347                 child.open();
348             }
349 
350             depth = -1;
351             child.innerDoms.content.scrollIntoView(true);
352             node.events.openStateChanged.splice(position, 1);
353         } else {
354             if(child !== null) {
355                 child.open();
356             }
357         }
358     };
359     var position = this.events.openStateChanged.push(evt) - 1;
360 
361     var firstId = path.shift();
362     this._openId(firstId);
363 };

377 JusTo.ui.Node.prototype._asyncOpen = function() {
378     var request = new XMLHttpRequest();
379     var objThis = this;
380     var icon = this.innerDoms.icon;
381     icon.className = "icon loading";
382 
383     request.onreadystatechange = function() {
384         if (request.readyState === 4 && request.status === 200) {
385             var result = JSON.parse(request.responseText).result;
386             result.forEach(function(entry) {
387                 var newNode = new JusTo.ui.Node(entry.value, entry.id, entry.urlSubItems);
388                 newNode.setTitle(entry.title);
389                 objThis.push(newNode);
390             });
391 
392             objThis.loaded = true;
393             objThis._setChildrenVisible(true);
394             objThis._nodeOpenChanged(objThis);
395             objThis._nodeAsyncOpened(objThis);
396         }
397     };
398     request.open("GET", this.urlSubItems, true);
399     request.send();
400 };

416 /**
417  * Select all children (not select current node, common used with root node)
418  *
419  * @this {Node}
420  * @param {boolean} flag
421  */
422 JusTo.ui.Node.prototype.selectAll = function(flag) {
423     if (flag === undefined)
424         flag = true;
425 
426     this.iterate(function(node, depth) {
427         if(node.selected == flag) return;
428         node.setSelected(flag, true);
429     });

432 /**
433  * Iterate all children recursively.
434  * Example: root.iterate(
435              function(node) {console.log(node.value)},
436              function(node) {return node.isLeaf()}
437             )
438  *
439  * @this {Node}
440  * @param {function} fnCallback A function to apply in each node found. 
441                                 You can use 'return false' in this function to stop iteration
442  * @param {function} fnFilter A function to determine if node will be iterate or not.
443  *                            Must return a boolean
444  * @see JusTo.ui.NodeFilter
445  */
446 JusTo.ui.Node.prototype.iterate = function(fnCallback, fnFilter) {
447     this._iterate(fnCallback, fnFilter, 0);
448 };
449 
450 JusTo.ui.Node.prototype._iterate = function(fnCallback, fnFilter, depth) {
451     ++depth;
452     var next = true;
453 
454     this.children.forEach(function(node) {
455         if (!fnFilter || (fnFilter(node))) {
456             if(fnCallback(node, depth) === false) {
457                 next = false;
458             }
459         }
460 
461         if(!next) return;
462         node._iterate(fnCallback, fnFilter, depth);
463     });
464 };
